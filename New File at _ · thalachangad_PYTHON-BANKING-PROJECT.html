def deposit():
print("*********************")
amount = float(input("Enter an amount to be deposited: "))
print("----------------------")
if amount < 0:
print("*********************")
print("That's not a valid amount")
print("*********************")
return 0
else:
return amount

def withdraw(balance):
print("*********************")
amount = float(input("Enter amount to be withdrawn: "))
print("*********************")

if amount > balance:
print("*********************")
print("Insufficient funds")
print("*********************")
return 0
elif amount < 0:
print("*********************")
print("Amount must be greater than 0")
print("*********************")
return 0
else:
return amount

def main():
balance = 0
is_running = True

while is_running:
print("*********************")
print(" Banking Program ")

if choice == '1':
show_balance(balance)
elif choice == '2':
balance += deposit()

balance -= withdraw(balance)

is_running = False

print("*********************")
